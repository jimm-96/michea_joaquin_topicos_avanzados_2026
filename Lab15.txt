CREATE TABLE Ventas (
    IdVenta NUMBER,
    ClienteID NUMBER,
    MontoVenta NUMBER,
    FechaRegistro DATE,
    CONSTRAINT pk_id_venta PRIMARY KEY (IdVenta)
)
PARTITION BY HASH (ClienteID)
PARTITIONS 4;

INSERT INTO Ventas (IdVenta, ClienteID, MontoVenta, FechaRegistro)
SELECT PedidoID, ClienteID, Total, FechaPedido
FROM Pedidos;
COMMIT;

EXPLAIN PLAN FOR
SELECT ClienteID, 
       COUNT(IdVenta) AS CantidadCompras,
       SUM(MontoVenta) AS SumaVentasTotales
FROM Ventas
GROUP BY ClienteID
ORDER BY SumaVentasTotales DESC;
