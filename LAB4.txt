DECLARE
    v_precio       NUMBER;
    v_producto_id  NUMBER := 2;
    v_bias         NUMBER := 500;      -- Umbral mínimo de precio
    precio_bajo    EXCEPTION;
BEGIN
    DBMS_OUTPUT.PUT_LINE('Verificando precio del producto ID: ' || v_producto_id);

    -- busca el precio del producto
    SELECT Precio INTO v_precio
    FROM Productos
    WHERE ProductoID = v_producto_id;

    DBMS_OUTPUT.PUT_LINE('Precio encontrado: $' || v_precio);

    -- Si el precio es menor al bias, lanza excepción
    IF v_precio < v_bias THEN
        RAISE precio_bajo;
    END IF;

    DBMS_OUTPUT.PUT_LINE('Precio válido.');

EXCEPTION
    WHEN precio_bajo THEN
        DBMS_OUTPUT.PUT_LINE('Error: El precio $' || v_precio || ' es menor al umbral permitido de $' || v_bias || '.');
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: No se encontró ningún producto con ID ' || v_producto_id || '.');
END;
/
