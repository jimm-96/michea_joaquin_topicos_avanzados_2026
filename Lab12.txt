-- ACTIVIDAD 1:
-- Crea una función calcular_total_con_descuento que reciba un PedidoID (parámetro IN) 
-- y devuelva el total del pedido con un descuento del 10% si el total supera 1000. 
-- Usa la función en un procedimiento aplicar_descuento_pedido que actualice el total del pedido.

CREATE OR REPLACE FUNCTION calcular_total_con_descuento(p_pedido_id IN NUMBER) RETURN NUMBER AS
    v_total_original NUMBER;
    v_total_con_descuento NUMBER;
BEGIN

    SELECT SUM(precio * cantidad)
    INTO v_total_original
    FROM DetallesPedido
    WHERE PedidoID = p_pedido_id;

    IF v_total_original > 1000 THEN
        v_total_con_descuento := v_total_original * 0.9;
    ELSE
        v_total_con_descuento := v_total_original;
    END IF;

    RETURN v_total_con_descuento;
END;
/
create or replace procedure aplicar_descuento_pedido(p_pedido_id IN NUMBER) AS
    v_total_con_descuento NUMBER;
BEGIN
    v_total_con_descuento := calcular_total_con_descuento(p_pedido_id);
    
    UPDATE Pedidos
    SET Total = v_total_con_descuento
    WHERE PedidoID = p_pedido_id;
END;
/

-- ACTIVIDAD 2:
-- Crea un trigger validar_cantidad_detalle que se dispare antes de insertar o 
-- actualizar en DetallesPedidos y verifique que la Cantidad sea mayor a 0. 
-- Si no, lanza un error.

CREATE OR REPLACE TRIGGER validar_cantidad_detalle
BEFORE INSERT OR UPDATE ON DetallesPedido
FOR EACH ROW
DECLARE
    v_cantidad NUMBER;
BEGIN
    v_cantidad := :NEW.Cantidad;

    IF v_cantidad <= 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'La cantidad debe ser mayor a 0.');
    END IF;
END;
/
