CREATE OR REPLACE FUNCTION calcular_edad_cliente(p_cliente_id IN NUMBER) RETURN NUMBER AS
    v_fecha_nac DATE;
    v_edad NUMBER;
BEGIN
    SELECT FechaNacimiento 
    INTO v_fecha_nac
    FROM Clientes
    WHERE ClienteID = p_cliente_id;
    
    v_edad := FLOOR(MONTHS_BETWEEN(SYSDATE, v_fecha_nac) / 12);
    
    RETURN v_edad;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20001, 'Cliente ID ' || p_cliente_id || ' no existe');
END calcular_edad_cliente;
/

--Pruebas
DECLARE
    v_edad NUMBER;
BEGIN
    v_edad := calcular_edad_cliente(1);
    DBMS_OUTPUT.PUT_LINE('Edad del cliente 1: ' || v_edad || ' años');
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END;        
/

DECLARE
    v_resultado NUMBER;
BEGIN
    v_resultado := calcular_edad_cliente(3);
    DBMS_OUTPUT.PUT_LINE('Cliente 3 tiene ' || v_resultado || ' años');
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Excepción capturada: ' || SQLERRM);
END;
/

CREATE OR REPLACE FUNCTION obtener_precio_promedio RETURN NUMBER AS
    v_promedio NUMBER;
BEGIN
    SELECT AVG(Precio)
    INTO v_promedio
    FROM Productos;
    
    RETURN v_promedio;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20002, 'No hay productos en la base de datos');
END obtener_precio_promedio;
/

SELECT ProductoID, Nombre, Precio
FROM Productos
WHERE Precio > obtener_precio_promedio()
ORDER BY Precio DESC;
/
