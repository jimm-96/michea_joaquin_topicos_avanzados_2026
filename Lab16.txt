CREATE INDEX idx_cli_ciudad_stgo ON Clientes(Ciudad);
CREATE INDEX idx_ped_fecha_rango ON Pedidos(FechaPedido);

EXPLAIN PLAN FOR
SELECT /*+ INDEX(cli idx_cli_ciudad_stgo) INDEX(ped idx_ped_fecha_rango) */
       cli.Nombre, COUNT(ped.PedidoID) AS CantidadPedidos
FROM Clientes cli
INNER JOIN Pedidos ped 
        ON cli.ClienteID = ped.ClienteID
WHERE cli.Ciudad = 'Santiago'
  AND ped.FechaPedido >= TO_DATE('2025-03-01', 'YYYY-MM-DD')
GROUP BY cli.Nombre;

--Optimiza la siguiente consulta para evitar un FULL TABLE SCAN en 
--DetallesPedidos y analiza el plan de ejecución antes y después de 
--la optimización.
--SELECT p.Nombre, SUM(dp.Cantidad * p.Precio) AS TotalVentas
--FROM Productos p, DetallesPedidos dp
--WHERE p.ProductoID = dp.ProductoID
--GROUP BY p.Nombre;

EXPLAIN PLAN FOR
SELECT prod.Nombre, SUM(det.Cantidad * prod.Precio) AS IngresosTotales
FROM Productos prod, DetallesPedidos det
WHERE prod.ProductoID = det.ProductoID
GROUP BY prod.Nombre;
SELECT * FROM TABLE(DBMS_XPLAN.DISPLAY);

CREATE INDEX idx_det_prod_fk ON DetallesPedidos(ProductoID);

EXPLAIN PLAN FOR
SELECT /*+ INDEX(det idx_det_prod_fk) */
       prod.Nombre, SUM(det.Cantidad * prod.Precio) AS IngresosTotales
FROM Productos prod
INNER JOIN DetallesPedidos det 
        ON prod.ProductoID = det.ProductoID
GROUP BY prod.Nombre;
SELECT * FROM TABLE(DBMS_XPLAN.DISPLAY);
